{
    bool IsSuccess
    T Data
    List<string>? Message
    string MessageSingle  // همه پیام‌ها به صورت یک رشته

    static ResultOperation<T> ToSuccessResult(T data)
    static ResultOperation<T> ToSuccessResult(string message, T data)
    static ResultOperation<T> ToSuccessResult(List<string> message, T data)
    static ResultOperation<T> ToFailedResult()

    static ResultOperation<T> ToFailedResult(List<string> message)
    static ResultOperation<T> ToFailedResult(string message, T data)
    static ResultOperation<T> ToFailedResult(List<string> message, T data)
}
